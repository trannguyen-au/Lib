the options available are:

	
$(document).ready(function() {
	$("#slider").wnMenuSlider({
		[option] : [value]
	});
});
	
OPTION - VALUE PAIRS :
defaults = {
	pagingType : "number", // "image" // "none" // need to have the appropriate css for the image paging type
	timerInterval: 6000,   // time to switch between each image
	animationSpeed: 500,   // time for fading animation
	effect: "fade",        // "none"
	autoRotate: true,
};
